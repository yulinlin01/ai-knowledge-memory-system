# JavaScript 手写题

## 1. 手写 Promise

### 题目
实现一个简易版 Promise，支持 `then`、`catch`、`finally`。

### 考点
- 状态管理（pending/fulfilled/rejected）
- then 链式调用
- 微任务调度

### 参考答案

```javascript
class MyPromise {
    constructor(executor) {
        this.state = 'pending';
        this.value = undefined;
        this.callbacks = [];

        const resolve = (value) => {
            if (this.state !== 'pending') return;
            this.state = 'fulfilled';
            this.value = value;
            this.callbacks.forEach(cb => cb.onFulfilled(value));
        };

        const reject = (reason) => {
            if (this.state !== 'pending') return;
            this.state = 'rejected';
            this.value = reason;
            this.callbacks.forEach(cb => cb.onRejected(reason));
        };

        try {
            executor(resolve, reject);
        } catch (e) {
            reject(e);
        }
    }

    then(onFulfilled, onRejected) {
        return new MyPromise((resolve, reject) => {
            const handle = (callback, resolve, reject) => {
                try {
                    const result = callback(this.value);
                    result instanceof MyPromise
                        ? result.then(resolve, reject)
                        : resolve(result);
                } catch (e) {
                    reject(e);
                }
            };

            if (this.state === 'fulfilled') {
                handle(onFulfilled, resolve, reject);
            } else if (this.state === 'rejected') {
                handle(onRejected, resolve, reject);
            } else {
                this.callbacks.push({
                    onFulfilled: () => handle(onFulfilled, resolve, reject),
                    onRejected: () => handle(onRejected, resolve, reject)
                });
            }
        });
    }

    catch(onRejected) {
        return this.then(null, onRejected);
    }

    finally(callback) {
        return this.then(
            value => MyPromise.resolve(callback()).then(() => value),
            reason => MyPromise.resolve(callback()).then(() => { throw reason; })
        );
    }
}
```

## 2. 手写深拷贝

### 题目
实现一个深拷贝函数，处理循环引用和特殊对象类型。

### 考点
- 递归实现
- 循环引用检测（WeakMap）
- 特殊对象处理（Date/RegExp/Map/Set）

### 参考答案

```javascript
function deepClone(obj, hash = new WeakMap()) {
    if (obj === null || typeof obj !== 'object') return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof RegExp) return new RegExp(obj);
    if (hash.has(obj)) return hash.get(obj);

    const cloneObj = Array.isArray(obj) ? [] : {};
    hash.set(obj, cloneObj);

    for (const key in obj) {
        if (obj.hasOwnProperty(key)) {
            cloneObj[key] = deepClone(obj[key], hash);
        }
    }
    return cloneObj;
}
```

## 3. 手写节流函数

### 题目
实现一个节流函数，控制函数执行频率。

### 考点
- 时间戳实现 vs 定时器实现
- 首次执行 / 最后一次执行
- 取消功能

### 参考答案

```javascript
function throttle(fn, delay) {
    let lastTime = 0;
    let timer = null;

    return function(...args) {
        const now = Date.now();

        if (now - lastTime >= delay) {
            lastTime = now;
            fn.apply(this, args);
        } else if (!timer) {
            timer = setTimeout(() => {
                lastTime = Date.now();
                fn.apply(this, args);
                timer = null;
            }, delay - (now - lastTime));
        }
    };
}

// 支持取消
throttle.cancel = () => {
    clearTimeout(timer);
    timer = null;
};
```
